var alien_types = [
    {code:10,level:0,points:5,crystal_drop:5},
    {code:10,level:1,points:10,crystal_drop:10},
    {code:11,level:0,points:15,crystal_drop:20},

    {code:16,level:0,points:40,crystal_drop:30}, // Barracudas
    {code:10,level:2,points:30,crystal_drop:35},
    {code:17,level:0,points:50,crystal_drop:45}, // Pointus
    {code:18,level:0,points:60,crystal_drop:60}, // Forks
    {code:19,level:0,points:100,crystal_drop:100}
  ]

this.tick = function(game) {
  if (game.step % 15 == 0)
  {
    var aliens = Math.max(5,Math.min(30,game.ships.length*2));
    if (game.aliens.length<aliens)
    {
      var minute = game.step/3600 ;
      var range = Math.min(alien_types.length,minute/4);
      var index = Math.floor(Math.random()*range);
      var angle = Math.random()*Math.PI*2;
      var alien = alien_types[index] ;
      alien.x = Math.cos(angle)*25 ;
      alien.y = Math.sin(angle)*25 ;
      game.addAlien(alien) ;
    }
    for (var i=0;i<game.ships.length;i++)
    {
      var ship = game.ships[i];
      if (ship.custom.radar_updated == null)
      {
        ship.custom.radar_updated = true ;
        ship.setUIComponent(radar_background);
      }
    }
  }
  if (game.step%905 == 0) {
    game.addCollectible({code:21,x:5,y:20});
    game.addCollectible({code:21,x:-11,y:-26});
    game.addCollectible({code:91,x:-17,y:20});
  }
  if (game.step%2605 == 0) {  
    game.addCollectible({code:12,x:-6,y:-6});
    game.addCollectible({code:91,x:5,y:-26});
  }
};
